fig, ax = plt.subplots(figsize=(8,8))  # Create a figure.
x = [1, 2, 3, 4] # Create some data.
y = [10, 24, 18, 120]
ax.plot(x,y, label='data', color='tomato', linewidth=8, linestyle='-');  # Plot some data.
ax.set_xlabel('Year')  # Add an x-label to the axes.
ax.set_ylabel('Number of cool people')# Add a y-label to the axes.
ax.set_title("Simple Line Plot")  # Add a title to the axes.
ax.set_xticks([1, 2, 3, 4])
ax.set_xlim(1,4)
plt.show()

fig, ax = plt.subplots(figsize=(8,8))  # Create a figure.

x = np.linspace(1,12,200) 
y1 = 0.6+0.05*x+np.random.uniform(0,0.2,200) # Create some data.
y2 = 0.6+0.02*x+np.random.uniform(0,0.2,200)
ax.scatter(x,y1, label='Germany', color='seagreen', s=1e2);  # Plot some data.
ax.scatter(x,y2, label='South Park', color='sandybrown', marker='o', s=1e2);
ax.set_xlabel('Age [yrs]')  # Add an x-label to the axes.
ax.set_ylabel('Height [m]')  # Add a y-label to the axes.
ax.set_title("Simple Scatter Plot")  # Add a title to the axes.
ax.set_ylim(0.8,1.4)
ax.set_xlim(2,12)
ax.legend()  # Add a legend.
plt.show()

fig, ax = plt.subplots(figsize=(8,8))  # Create a figure containing a single axes.
x1 = np.random.normal(1.6, 0.1, 1000) # Create some data
x2 = np.random.normal(1.2, 0.1, 1000) # Create some data
ax.hist(x1, bins=10, range=(1.1,2.1), label='Germany', facecolor='white', edgecolor='seagreen', linewidth=4);  # Plot some data on the axes.
ax.hist(x2, bins=10, range=(0.7,1.7), label='South Park', facecolor='white', edgecolor='sandybrown', linewidth=4, alpha=1);  # Plot some data on the axes.
ax.set_xlabel('Height [m]')  # Add an x-label to the axes.
ax.set_ylabel('Count')  # Add a y-label to the axes.
ax.set_title("Simple Histogram")  # Add a title to the axes.
ax.legend(loc='upper right')  # Add a legend.
ax.set_ylim(0,420)
plt.show()
